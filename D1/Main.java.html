import java.math.BigInteger;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        BigInteger b = new BigInteger("1234");
        System.out.println(b.isProbablePrime(1));
        System.out.println(b.nextProbablePrime());
        System.out.println(factorial(a));
        System.out.println(fibonacci(a));
        for (int i = 0; i <= a; i++) {
            System.out.print(fibonacci(i)+" ");
        }
    }
    static BigInteger factorial(int a){

        BigInteger x = new BigInteger("1");
        for (int i = 1; i < a; i++) {
            BigInteger g = BigInteger.valueOf(i);
            x = x.multiply(g);
        }
        return x;
    }
    static BigInteger fibonacci(int a){
        if (a == 0){
            return new BigInteger("0");
        }
        if (a == 1){
            return new BigInteger("1");
        }
        BigInteger n1 = new BigInteger("0");
        BigInteger n2 = new BigInteger("1");
        for (int i = 0; i < a-1; i++) {
            BigInteger temp = n1.add(n2);
            n1 = n2;
            n2 = temp;
        }
        return n2;
    }
}
